\newglossaryentry{rnn}{
	name=RNN,
	description={Recurrent Neural Network.}
}

\newglossaryentry{cnn}{
	name=CNN,
	description={Convolutional Neural Network.}
}

\newglossaryentry{lstm}{
	name=LSTM,
	description={Long Short-Term Memory. Type of recurrent unit, which deals with the vanishing gradient problem.}
}

\newglossaryentry{gru}{
	name=GRU,
	description={Gated Recurrent Unit. Type of recurrent unit based on \gls{lstm}.}
}

\newglossaryentry{relu}{
	name=ReLU,
	description={Rectified Linear Unit. Type of unit combining linear unit and rectifier as nonlinearity.}
}

\newglossaryentry{coco}{
	name=MS COCO,
	description={Microsoft Common Objects in Context. Computer vision dataset and popular captioning and detection challenge.}
}

\newglossaryentry{mert}{
	name=MERT,
	description={Minimum Error Rate Training. Method used for re-ranking captions based on several features.}
}

\newglossaryentry{sgd}{
	name=SGD,
	description={Stochastic Gradient Descent. Numeric method for achieving a minimum of function. Used in neural network training.}
}

\newglossaryentry{bleu}{
	name=BLEU,
	description={Bilingual Evaluation Understudy. Evaluation metric for machine translation.}
}

\newglossaryentry{meteor}{
	name=METEOR,
	description={Metric for Evaluation of Translation with Explicit Ordering. Evaluation metric for machine translation. Enhanced \gls{bleu}}
}

\newglossaryentry{cider}{
	name=CIDEr,
	description={Consensus-based Image Description Evaluation. Evaluation metric for image captions.}
}

\newglossaryentry{vgg}{
	name=VGGnet,
	description={16-layer \gls{cnn}~\cite{DBLP:journals/corr/SimonyanZ14a} trained for classifying ImageNet images.}
}

\newglossaryentry{dmsm}{
	name=DMSM,
	description={Deep Multimodal Similarity Model. Score measuring similarity between images and text.}
}

\newglossaryentry{lrcn}{
	name=\emph{LRCN},
	description={Long-term Recurrent Convolutional Networks. Image captioning model~\cite{DBLP:journals/corr/DonahueHGRVSD14}.}
}
